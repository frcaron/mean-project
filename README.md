
# MEAN-PROJECT

Variable env :
	- BDD_URL          (default : 'mongodb://localhost/mean-project')
	- SECRET           (default : 'secretmeanproject')
	- LEVEL_LOG        (default : 'debug')
	- HOSTNAME         (default : 'localhost')
	- PORT             (default : 3000)
	- FB_CLIENT_ID     (no default)
	- FB_CLIENT_SECRET (no default)
